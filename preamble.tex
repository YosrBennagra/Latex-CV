% Header macro to render name (uppercase bold) on left and role on right same baseline
% Uses simple hbox with stretchable space for ATS-safe linear text ordering.
\newcommand{\makecvheader}{%
  \begingroup
  \par\vspace*{1pt}
  {\Large\textbf{\MakeUppercase{\CVName}}\hfill\Large\textbf{\MakeUppercase{\CVRole}}}\par
  \vspace{1pt}
  \endgroup
}
